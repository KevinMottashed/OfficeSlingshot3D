00115 {
00116     if (input.size() < sizeof(PacketHeader<PacketType>))
00117     {
00118         // all data packets need to have a least a header
00119         return false;
00120     }
00121 
00122     // grab the header from the packet
00123     PacketHeader<PacketType>* tempHeader = (PacketHeader<PacketType>*) &input[0];
00124 
00125     // the size of the entire packet (including the header)
00126     unsigned int size = sizeof(PacketHeader<PacketType>) + tempHeader->size;
00127 
00128     // the stream needs to be bigger or equal to the packet size for the packet to be complete
00129     if (size <= input.size())
00130     {
00131         // we will use this counter to navigate through the stream of bytes
00132         unsigned int dataIndex = 0;
00133 
00134         // clear the packet
00135         clear();
00136 
00137         // add the header to the packet
00138         header->insert(header->end(), &input[dataIndex], &input[dataIndex] + sizeof(PacketHeader<PacketType>));
00139         dataIndex += sizeof(PacketHeader<PacketType>);
00140 
00141         // some packets are headers only, make sure there is data before copying it into the packet
00142         if (tempHeader->size > 0)
00143         {
00144             // add the data to the packet
00145             data->insert(data->end(), &input[dataIndex], &input[dataIndex] + tempHeader->size);
00146             dataIndex += tempHeader->size;
00147         }
00148 
00149         // erase this packet from the input stream
00150         input.erase(input.begin(), input.begin() + size);
00151 
00152         return true;
00153     }
